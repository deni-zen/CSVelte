 22 /**
 23  * Table row abstract base class
 24  * Represents a row of tabular data (represented by CSVelte\Table\Data objects)
 25  *
 26  * @package CSVelte
 27  * @subpackage CSVelte\Table
 28  * @since v0.1
 29  * @todo On all of the ArrayAccess methods, the docblocks say that $offset can be
 30  *     either an integer offset or a string index, but that isn't true, they must
 31  *     be an integer offset. Fix docblocks.
 32  */
 33 abstract class AbstractRow implements Iterator, Countable, ArrayAccess
 34 {
 35     /**
 36      * @var array The columns within the row
 37      * @todo Technically a row doesn't contain "columns". It contains "data" or
 38      *     "datum" if we want to really get down to semantics. But I'm not sure
 39      *     whether that would be a better name for this or not...
 40      */
 41     protected $columns;

 43     /**
 44      * @var integer The current position within $columns
 45      */
 46     protected $position;
 47 
 48     /**
 49      * @var HeaderRow The header row
 50      */
 51     protected $headers;

 53     /**
 54      * @var CSVelte\Flavor The flavor for this row
 55      * @todo I really don't want to have every object in the libary carry around
 56      *     a Flavor object. In fact, I would prefer it if only one did. The Reader
 57      *     or the Writer or whatever the action I'm performing's main class is.
 58      *     I need to find a solution that allows me to keep the Flavor in one
 59      *     place and have anything else that needs to know about it, either NOT
 60      *     need to know about it, or get it from somewhere else (rather than )
 61      *     storing it, itself).
 62      */
 63     protected $flavor;
 64 
 65     /**
 66      * Class constructor
 67      *
 68      * @param array An array (or anything that looks like one) of data (columns)
 69      * @param CSVelte\Flavor
 70      * @access public
 71      * @todo This doesn't really need to be passed $flavor
 72      * @todo Look into SplFixedArray for csv sources w/out a header row.
 73      */
 74     public function __construct(array $columns, Flavor $flavor = null)
 75     {
 76         if (is_null($flavor)) $flavor = new Flavor;
 77         $this->flavor = $flavor;
 78         $this->columns = array_values($columns);
 79         $this->rewind();
 80     }

 82     /**
 83      * String overloading
 84      *
 85      * @return string representation of this object
 86      * @access public
 87      * @todo Should this end with a lineTerminator? I think it probably should...
 88      */
 89     public function __toString()
 90     {
 91         return $this->join();
 92     }

 94     /**
 95      * Join columns together using specified delimiter
 96      *
 97      * @param char The delimiter character
 98      * @return string
 99      * @access public
100      */
101     public function join($delimiter = null)
102     {
103         if (is_null($delimiter)) $delimiter = $this->flavor->delimiter;
104         return implode($delimiter, $this->columns);
105     }

133     /**
134      * Get the current column's data object
135      *
136      * @return CSVelte\Table\Data
137      * @access public
138      */
139     public function current()
140     {
141         if (!array_key_exists($this->position, $this->columns)) {
142             throw new \OutOfBoundsException("Undefined index: " . $this->position);
143         }
144         return $this->columns[$this->position];
145     }
146 
147     /**
148      * Get the current key (column number or header, if available)
149      *
150      * @return string The "current" key
151      * @access public
152      * @todo Figure out if this can return a CSVelte\Table\HeaderData object so long as it
153      *     has a __toString() method that generated the right key...
154      */
155     public function key()
156     {
157         return isset($this->headers[$this->position]) ? $this->headers[$this->position] : $this->position;
158     }
159 
160     /**
161      * Advance the internal pointer to the next column's data object
162      * Also returns the next column's data object if there is one
163      *
164      * @return CSVelte\Table\Data The "next" column's data
165      * @access public
166      */
167     public function next()
168     {
169         $this->position++;
170         if ($this->valid()) return $this->current();
171     }
172 
173     /**
174      * Return the internal pointer to the first column and return that object
175      *
176      * @return void
177      * @access public
178      */
179     public function rewind()
180     {
181         $this->position = 0;
182         if ($this->valid()) return $this->current();
183     }
184 
185     /**
186      * Is the current position within the row's data columns valid?
187      *
188      * @return boolean
189      * @access public
190      */
191     public function valid()
192     {
193         return array_key_exists($this->position, $this->columns);
194     }

205     public function offsetExists($offset)
206     {
207         try {
208             Utils::array_get($this->columns, $offset, null, true);

210             // now check $this->properties?
211             return false;
212         }
213         return true;

246      *
247      * @param integer|string Offset/index

251      * @todo I'm not sure if these objects will stay immutable or not yet...
252      */
253     public function offsetUnset($offset)
254     {
255         $this->raiseImmutableException();
256     }

258     /**
259      * Throw exception unless offset/index exists
260      *
261      * @param integer|string Offset/index
262      * @return void
263      * @access protected
264      * @throws \OutOfBoundsException
265      */
266     protected function assertOffsetExists($offset)
267     {
268         if (!$this->offsetExists($offset)) {
269             throw new \OutOfBoundsException("Undefined offset: " . $offset);
270         }
271     }

273     /**
274      * Raise (throw) immutable exception
275      *
276      * @param string Message
277      * @return void
278      * @access protected
279      * @throws CSVelte\Exception\ImmutableException
280      */
281     protected function raiseImmutableException($msg = null)
282     {
283         // columns are immutable, cannot be set
284         throw new ImmutableException($msg ?: 'Cannot change immutable column data');
285     }
