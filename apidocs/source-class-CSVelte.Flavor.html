 20 /**
 21  * CSV Flavor
 22  *
 23  * Represents a particular "flavor"  of CSV. Inspired by python's csv "dialects".
 24  * Also inspired by Frictionless Data's "dialect description" format and the W3C's
 25  * CSV on the Web Working Group and their work on CSV dialects.
 26  *
 27  * @package CSVelte
 28  * @subpackage Flavor
 29  * @since v0.1
 30  * @property-read string $delimiter The delimiter character
 31  * @property-read string $quoteChar The quoting character
 32  * @property-read string $lineTerminator The character sequence used to terminate rows of data
 33  * @property-read string $escapeChar The character used to escape quotes within a quoted string
 34  *     Mutually exclusive to $doubleQuote
 35  * @property-read bool $doubleQuote If true, quote characters will be escaped by preceding them
 36  *     with another quote character. Mutually exclusive to $escapeChar
 37  * @property-read string $quoteStyle One of four class constants that determine which cells are quoted
 38  * @property-read bool $header If true, first row should be treated as a header row
 39  */
 40 class Flavor

137     /**
138      * Class constructor
139      *
140      * The attributes that make up a flavor object can only be specified by
141      * passing them in an array as key => value pairs to the constructor. Once
142      * the flavor object is created, its attributes cannot be changed.
143      *
144      * @param array The attributes that define this particular flavor. These
145      *     attributes are immutable. They can only be set here.
146      */
147     public function __construct($attributes = null)
148     {
149         if (!is_null($attributes)) {
150             if (!is_array($attributes)) {
151                 // @todo throw exception?
152                 return;
153             }
154             foreach ($attributes as $attr => $val) {
155                 $this->assertValidAttribute($attr);
156                 $this->$attr = $val;
157             }
158         }
159     }

161     /**
162      * Does this flavor of CSV have a header row?
163      *
164      * The difference between $flavor->header and $flavor->hasHeader() is that
165      * hasHeader() is always going to give you a boolean value, whereas
166      * $flavor->header may be null. A null value for header could mean that the
167      * taster class could not reliably determine whether or not there was a
168      * header row or it could simply mean that the flavor was instantiated with
169      * no value for the header property.
170      *
171      * @param void
172      * @return boolean
173      * @access public
174      */
175     public function hasHeader()
176     {
177         return (bool) $this->header;
178     }

180     /**
181      * Assert valid attribute name.
182      * Assert that a particular attribute is valid (basically just that it exists)
183      * and throw an exception otherwise
184      *
185      * @param string The attribute to check validity of
186      * @return void
187      * @access protected
188      * @throws UnknownAttributeException
189      * @internal
190      * @todo This should accept a second parameter for value that asserts the value
191      *     is a valid value
192      */
193     protected function assertValidAttribute($attr)
194     {
195         if (!property_exists(self::class, $attr))
196             throw new UnknownAttributeException("Unknown attribute: " . $attr);
197     }

199     /**
200      * Copy this flavor object
201      *
202      * Because flavor attributes are immutable, it is implossible to change their
203      * attributes. If you need to change a flavor's attributes, call this method
204      * instead, specifying which attributes are to be changed.
205      *
206      * @param array An array of attribute name/values to change in the copied flavor
207      * @return CSVelte\Flavor A flavor object with your new attributes
208      * @access public
209      * @todo I may want to remove the array type-hint so that this can accept
210      *     array-like objects and iterables as well. Not sure...
211      */
212     public function copy(array $attribs = array())
213     {
214         $attributes = get_object_vars($this);
215         return new Flavor(array_merge($attributes, $attribs));
216     }
