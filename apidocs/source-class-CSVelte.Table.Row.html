27 class Row extends AbstractRow
28 {
29     /**
30      * @var array Same as columns only indexed using headers rather than numbers
31      */
32     protected $assocCols = array();
33 
34     /**
35      * Set the header row (so that it can be used to index the row)
36      *
37      * @param CSVelte\Table\HeaderRow
38      * @return void (return $this?)
39      * @access public
40      */
41     public function setHeaderRow(AbstractRow $headers)
42     {
43         if (!($headers instanceof HeaderRow)) {
44             $headers = new HeaderRow($headers->toArray());
45         }
46         $headerArray = $headers->toArray();
47         if (($hcount = $headers->count()) !== ($rcount = $this->count())) {
48             if ($hcount > $rcount) {
49                 // header count is long, could be an error, but lets just fill in the short row with null values...
50                 $this->columns = array_pad($this->columns, $hcount, null);
51             } else {
52                 // @todo This is too strict. I need a way to recover from this a little better...
53                 // header count is short, this is likely an error...
54                 throw new InvalidHeaderException("Header count ({$hcount}) does not match column count ({$rcount}).");
55             }
56         }
57         $this->headers = $headers;
58         $this->assocCols = array_combine($headerArray, $this->columns);
59     }
60 
61     /**
62      * Is there an offset at specified position?
63      *
64      * @param integer Offset
65      * @return boolean
66      * @access public
67      */
68     public function offsetExists($offset)
69     {
70         try {
71             Utils::array_get($this->assocCols, $offset, null, true);
72         } catch (\OutOfBoundsException $e) {
73             // now check $this->properties?
74             return parent::offsetExists($offset);
75         }
76         return true;
77     }
78 
79     /**
80      * Retrieve data at specified column offset
81      *
82      * @param integer Offset
83      * @return CSVelte\Table\Data
84      * @access public
85      */
86     public function offsetGet($offset)
87     {
88         try {
89             $val = Utils::array_get($this->assocCols, $offset, null, true);
90         } catch (\OutOfBoundsException $e) {
91             return parent::offsetGet($offset);
92         }
93         return $val;
94     }
